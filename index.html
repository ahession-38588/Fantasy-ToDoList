<script>

  document.addEventListener("DOMContentLoaded", function () {

    function toggleMenu() {

      console.log("Toggle menu clicked");

      const menu = document.getElementById("menuDropdown");

      if (menu) {

        menu.style.display = menu.style.display === "block" ? "none" : "block";

        console.log("Menu display set to:", menu.style.display);

      } else {

        console.error("menuDropdown not found in DOM");

      }

    }


    function loginUser() {

      const username = document.getElementById("username").value.trim();

      console.log("Login attempt with username:", username);

      if (!username) {

        alert("Please enter your name to begin your adventure.");

        return;

      }

      localStorage.setItem("adventurerName", username);

      document.getElementById("loginScreen").style.display = "none";

      document.getElementById("startScreen").style.display = "flex";

      console.log("Login successful, startScreen displayed");

    }


    function goHome() {

      console.log("Returning to home screen");

      document.getElementById("loginScreen").style.display = "flex";

      document.getElementById("startScreen").style.display = "none";

      document.getElementById("questLog").style.display = "none";

    }


    window.toggleMenu = toggleMenu;

    window.loginUser = loginUser;

    window.goHome = goHome;

    window.beginQuest = function() {

      console.log("Begin Quest button clicked");

      document.getElementById("startScreen").style.display = "none";

      document.getElementById("questLog").style.display = "flex";

    };


    window.addQuest = function() {

      const input = document.getElementById("taskInput");

      const text = input.value.trim();

      if (text === "") {

        alert("Please enter a quest!");

        return;

      }

      const li = document.createElement("li");

      li.textContent = text;

      document.getElementById("taskList").appendChild(li);

      input.value = "";

    };

  });

</script>
